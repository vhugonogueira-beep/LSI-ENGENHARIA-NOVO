                        <button onClick={()=>{setProjetoSel(p.id);setObraTab("resumo");}} style={{...S.btn,padding:"5px 12px",fontSize:11,background:T.bg4,color:T.blue,border:`1px solid ${T.blue}30`}}>Abrir →</button>
                      </td>
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </div>
        </div>
      </div>
    );
  };

  // ════════════════════════════════════════════════════════════════════
  // TAB: FORNECEDORES
  // ════════════════════════════════════════════════════════════════════
  const TabFornecedores = () => (
    <div>
      <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:20}}>
        <div style={{fontSize:20,fontWeight:900,color:T.blue,letterSpacing:"-0.02em"}}>🏢 Fornecedores</div>
        <button onClick={()=>{setEditForn(null);setFornForm({nome:"",tipo:"Negociado"});setShowFornModal(true);}} style={S.btn}>+ Adicionar</button>
      </div>
      <div style={{display:"grid",gridTemplateColumns:"repeat(auto-fill,minmax(260px,1fr))",gap:12}}>
        {fornecedores.map(f=>{
          const tc={Próprio:T.green,Negociado:T.blue,Mercado:T.amber,Subcontratado:T.purple}[f.tipo]||T.txMut;
          return(
            <div key={f.id} style={{...S.card}}>
              <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start"}}>
                <div>
                  <div style={{fontSize:15,fontWeight:700,color:T.txPri,marginBottom:6}}>{f.nome}</div>
                  <span style={{background:tc+"18",color:tc,border:`1px solid ${tc}40`,padding:"3px 10px",borderRadius:6,fontSize:11,fontWeight:600}}>{f.tipo}</span>
                </div>
                <div style={{display:"flex",gap:6}}>
                  <button onClick={()=>{setEditForn(f);setFornForm({nome:f.nome,tipo:f.tipo});setShowFornModal(true);}} style={{...S.ghost,padding:"5px 10px",fontSize:11}}>✏️</button>
                  <button onClick={()=>setFornecedores(prev=>prev.filter(x=>x.id!==f.id))} style={{...S.ghost,color:T.red,border:`1px solid ${T.red}30`,padding:"5px 10px",fontSize:11}}>🗑️</button>
                </div>
              </div>
            </div>
          );
        })}
      </div>
      {showFornModal&&(
        <div style={{position:"fixed",inset:0,background:"#000000b0",display:"flex",alignItems:"center",justifyContent:"center",zIndex:500}}>
          <div style={{background:T.bg2,borderRadius:14,border:`1px solid ${T.brBase}`,padding:26,width:380}}>
            <div style={{fontSize:15,fontWeight:800,color:T.blue,marginBottom:18}}>{editForn?"✏️ Editar Fornecedor":"➕ Novo Fornecedor"}</div>
            <div style={{marginBottom:12}}><label style={S.label}>Nome</label>
              <input value={fornForm.nome} onChange={e=>setFornForm(p=>({...p,nome:e.target.value}))} style={S.input} placeholder="Nome do fornecedor"/></div>
            <div style={{marginBottom:20}}><label style={S.label}>Tipo de Contrato</label>
              <select value={fornForm.tipo} onChange={e=>setFornForm(p=>({...p,tipo:e.target.value}))} style={S.input}>
                {["Próprio","Negociado","Mercado","Subcontratado"].map(t=><option key={t}>{t}</option>)}</select></div>
            <div style={{display:"flex",gap:10}}>
              <button onClick={()=>setShowFornModal(false)} style={{...S.ghost,flex:1,padding:10}}>Cancelar</button>
              <button onClick={()=>{
                if(!fornForm.nome.trim())return;
                if(editForn) setFornecedores(prev=>prev.map(f=>f.id===editForn.id?{...f,...fornForm}:f));
                else setFornecedores(prev=>[...prev,{id:Date.now().toString(),...fornForm}]);
                setShowFornModal(false);setEditForn(null);
              }} style={{...S.btn,flex:1,padding:10}}>{editForn?"Salvar":"Adicionar"}</button>
            </div>
          </div>
        </div>
      )}
    </div>
  );

  // ════════════════════════════════════════════════════════════════════
  // TAB: RESUMO
  // ════════════════════════════════════════════════════════════════════
  const TabResumo = () => {
    const byCategory = CATEGORIAS.map(cat => {
      const items = orcItems.filter(i=>i.resumo===cat);
      const base  = items.reduce((s,i)=>s+i.vl_custom*i.qtde,0);
      return {cat, items, base, final: base*(1-discount/100)*(1+bdi/100)*(1+lucro/100)};
    }).filter(x=>x.items.length>0);

    return (
      <div>
        <div style={{fontSize:20,fontWeight:900,color:T.blue,marginBottom:20,letterSpacing:"-0.02em"}}>📈 Resumo do Orçamento</div>
        {orcItems.length===0?(
          <div style={{...S.card,textAlign:"center",padding:60}}>
            <div style={{fontSize:36,marginBottom:10}}>📊</div>
            <div style={{color:T.txMut,fontSize:14}}>Adicione itens na aba Orçamento para ver o resumo</div>
          </div>
        ):(
          <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:16}}>
            <div style={S.card}>
              <div style={{fontSize:13,fontWeight:700,color:T.txPri,marginBottom:14}}>💰 Composição de Custos</div>
              {[
                {l:"Custo Mercado (Médio)",v:totalMercado,c:T.txMut},
                {l:"Custo Base LS Office",v:totalCustom,c:T.txSec},
                {l:`(-) Desconto ${discount}%`,v:totalCustom*discount/100,c:T.red},
                {l:"= Após Desconto",v:totalCustom*(1-discount/100),c:T.txPri,bold:true},
                {l:`(+) BDI ${bdi}%`,v:totalCustom*(1-discount/100)*bdi/100,c:T.purple},
                {l:`(+) Lucro ${lucro}%`,v:totalCustom*(1-discount/100)*(1+bdi/100)*lucro/100,c:T.green},
                {l:"= TOTAL FINAL",v:totalFinal,c:T.amber,bold:true,big:true},
              ].map(({l,v,c,bold,big})=>(
                <div key={l} style={{display:"flex",justifyContent:"space-between",padding:"8px 0",borderBottom:`1px solid ${T.brSub}`}}>
                  <span style={{fontSize:12,color:bold?T.txSec:T.txMut}}>{l}</span>
                  <span style={{fontSize:big?17:13,fontWeight:bold?900:600,color:c}}>{fmt(v)}</span>
                </div>
              ))}
            </div>
            <div style={S.card}>
              <div style={{fontSize:13,fontWeight:700,color:T.txPri,marginBottom:14}}>📦 Por Categoria</div>
              {byCategory.map(({cat,items,base,final})=>(
                <div key={cat} style={{marginBottom:12}}>
                  <div style={{display:"flex",justifyContent:"space-between",marginBottom:4}}>
                    <span style={{fontSize:12,fontWeight:600,color:T.txSec}}>{cat} <span style={{color:T.txMut,fontWeight:400}}>({items.length})</span></span>
                    <span style={{fontSize:12,fontWeight:700,color:T.amber}}>{fmt(final)}</span>
                  </div>
                  <div style={{height:5,borderRadius:3,background:T.bg4,overflow:"hidden"}}>
                    <div style={{height:"100%",width:`${pct(base,totalCustom)}%`,background:T.blue,borderRadius:3}}/>
                  </div>
                </div>
              ))}
            </div>
            <div style={{...S.card,gridColumn:"span 2"}}>
              <div style={{fontSize:13,fontWeight:700,color:T.txPri,marginBottom:14}}>📋 Itens Selecionados</div>
              <div style={{overflowX:"auto"}}>
                <table style={{width:"100%",borderCollapse:"collapse",fontSize:12}}>
                  <thead><tr style={{borderBottom:`1px solid ${T.brBase}`}}>
                    {["Código","Categoria","Solução","Config.","Unid","Qtde","VL Unit","Subtotal","Desc%","BDI%","Lucro%","Total Final"].map(h=>(
                      <th key={h} style={{padding:"7px 8px",textAlign:"left",color:T.txMut,fontWeight:700,fontSize:10,letterSpacing:"0.05em",whiteSpace:"nowrap"}}>{h}</th>
                    ))}
                  </tr></thead>
                  <tbody>
                    {orcItems.map((item,i)=>(
                      <tr key={item.cod} style={{borderBottom:`1px solid ${T.brSub}`,background:i%2===0?"transparent":T.bg1+"50"}}>
                        <td style={{padding:"7px 8px",color:T.blue,fontWeight:700}}>{item.cod}</td>
                        <td style={{padding:"7px 8px"}}><span style={{background:T.bg4,color:T.txSec,padding:"2px 6px",borderRadius:4,fontSize:10}}>{item.resumo}</span></td>
                        <td style={{padding:"7px 8px",color:T.txSec,maxWidth:160,overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap"}} title={item.solucao}>{item.solucao}</td>
                        <td style={{padding:"7px 8px",color:T.txMut,maxWidth:100,overflow:"hidden",textOverflow:"ellipsis",whiteSpace:"nowrap"}}>{item.config||"—"}</td>
                        <td style={{padding:"7px 8px",color:T.txMut}}>{item.unid}</td>
                        <td style={{padding:"7px 8px",color:T.txPri,fontWeight:600}}>{item.qtde}</td>
                        <td style={{padding:"7px 8px",color:T.green,fontWeight:600,whiteSpace:"nowrap"}}>{fmt(item.vl_custom)}</td>
                        <td style={{padding:"7px 8px",color:T.txPri,whiteSpace:"nowrap"}}>{fmt(item.vl_custom*item.qtde)}</td>
                        <td style={{padding:"7px 8px",color:T.red}}>{item.descItem!==undefined?`${item.descItem}%`:`${discount}%`}</td>
                        <td style={{padding:"7px 8px",color:T.purple}}>{item.bdiItem!==undefined?`${item.bdiItem}%`:`${bdi}%`}</td>
                        <td style={{padding:"7px 8px",color:T.green}}>{item.lucroItem!==undefined?`${item.lucroItem}%`:`${lucro}%`}</td>
                        <td style={{padding:"7px 8px",color:T.amber,fontWeight:700,whiteSpace:"nowrap"}}>{fmt(calcItemFinal(item))}</td>
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
            </div>
          </div>
        )}
      </div>
    );
  };

  // ════════════════════════════════════════════════════════════════════
  // TAB: HISTÓRICO
  // ════════════════════════════════════════════════════════════════════
  const TabHistorico = () => (
    <div>
      <div style={{fontSize:20,fontWeight:900,color:T.blue,marginBottom:20,letterSpacing:"-0.02em"}}>📁 Histórico de Orçamentos</div>
      {historico.length===0?(
        <div style={{...S.card,textAlign:"center",padding:60}}>
          <div style={{fontSize:36,marginBottom:10}}>📂</div>
          <div style={{color:T.txMut,fontSize:14}}>Nenhum orçamento salvo ainda</div>
          <div style={{color:T.txDis,fontSize:12,marginTop:6}}>Crie itens na aba Orçamento e clique em "💾 Salvar"</div>
        </div>
      ):(
        <div style={{display:"flex",flexDirection:"column",gap:12}}>
          {historico.map(orc=>(
            <div key={orc.id} style={S.card}>
              <div style={{display:"flex",justifyContent:"space-between",alignItems:"flex-start",marginBottom:10}}>
                <div>
                  <div style={{display:"flex",gap:8,alignItems:"center",marginBottom:6,flexWrap:"wrap"}}>
                    <span style={{fontSize:14,fontWeight:800,color:T.blue}}>{orc.id}</span>
                    <span style={{background:T.bg3,color:T.txMut,padding:"2px 9px",borderRadius:5,fontSize:10}}>{orc.area}</span>
                    <span style={{fontSize:10,color:T.txDis}}>{orc.data}</span>
                  </div>
                  {orc.siteInfo?.siteId&&(
                    <div style={{display:"flex",gap:12,flexWrap:"wrap"}}>
                      {[{l:"Site",v:orc.siteInfo.siteId},{l:"Detentora",v:orc.siteInfo.nomeDetentora},{l:"Operadora",v:orc.siteInfo.operadora},{l:"Município",v:orc.siteInfo.municipio},{l:"UF",v:orc.siteInfo.uf}].filter(x=>x.v).map(({l,v})=>(
                        <span key={l} style={{fontSize:11}}><span style={{color:T.txMut}}>{l}: </span><span style={{color:T.txSec,fontWeight:600}}>{v}</span></span>
                      ))}
                    </div>
                  )}
                </div>
                <div style={{textAlign:"right",flexShrink:0,marginLeft:16}}>
                  <div style={{fontSize:20,fontWeight:900,color:T.amber}}>{fmt(orc.totalFinal)}</div>
                  <div style={{fontSize:11,color:T.txMut}}>Desc {orc.discount}% · BDI {orc.bdi}% · Lucro {orc.lucro}%</div>
                </div>
              </div>
              <div style={{borderTop:`1px solid ${T.brSub}`,paddingTop:10}}>
                <div style={{fontSize:11,color:T.txMut,marginBottom:6}}>{orc.itens.length} item(ns):</div>
                <div style={{display:"flex",flexWrap:"wrap",gap:6}}>
                  {orc.itens.map(item=>(
                    <span key={item.cod} style={{background:T.bg3,border:`1px solid ${T.brBase}`,borderRadius:6,padding:"3px 10px",fontSize:11,color:T.txSec}}>
                      {item.cod} <span style={{color:T.txMut}}>×{item.qtde}</span>
                    </span>
                  ))}
                </div>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );

  // ════════════════════════════════════════════════════════════════════
  // SITE INFO MODAL
  // ════════════════════════════════════════════════════════════════════
  const SiteModal = () => (
    <div style={{position:"fixed",inset:0,background:"#000000b0",display:"flex",alignItems:"center",justifyContent:"center",zIndex:600,padding:16}}>
      <div style={{background:T.bg2,borderRadius:14,border:`1px solid ${T.brBase}`,padding:26,width:500,maxWidth:"100%"}}>
        <div style={{fontSize:15,fontWeight:800,color:T.blue,marginBottom:20}}>📍 Informações do Site</div>
        <div style={{display:"grid",gridTemplateColumns:"1fr 1fr",gap:12,marginBottom:12}}>
          {[{l:"ID do Site",k:"siteId",ph:"Ex: PAPCJ001"},{l:"Operadora",k:"operadora",ph:"Ex: Vivo"},{l:"Nome da Detentora",k:"nomeDetentora",ph:"Ex: Highline"},{l:"Site Detentora",k:"siteDetentora",ph:"Ex: HIG-001"},{l:"UF",k:"uf",ph:"SP"},{l:"Município",k:"municipio",ph:"Ex: Campinas"}].map(({l,k,ph})=>(
            <div key={k}><label style={S.label}>{l}</label>
              <input value={siteInfo[k]} onChange={e=>setSiteInfo(p=>({...p,[k]:e.target.value}))} style={S.input} placeholder={ph}/></div>
          ))}
          <div style={{gridColumn:"span 2"}}><label style={S.label}>Endereço Completo</label>
            <input value={siteInfo.endereco} onChange={e=>setSiteInfo(p=>({...p,endereco:e.target.value}))} style={S.input} placeholder="Rua, número, bairro..."/></div>
        </div>
        <div style={{display:"flex",gap:10}}>
          <button onClick={()=>setShowSiteModal(false)} style={{...S.ghost,flex:1,padding:10}}>Cancelar</button>
          <button onClick={()=>setShowSiteModal(false)} style={{...S.btn,flex:1,padding:10}}>Confirmar</button>
        </div>
      </div>
    </div>
  );

  // ════════════════════════════════════════════════════════════════════
  // RENDER MAIN
  // ════════════════════════════════════════════════════════════════════
  return (
    <div style={{fontFamily:"'Inter','DM Sans',system-ui,sans-serif",minHeight:"100vh",background:T.bg0,color:T.txPri,display:"flex",fontSize:13}}>
      <Sidebar/>
      <div style={{flex:1,overflowY:"auto",padding:24}}>
        {tab==="orcamento"    && <TabOrcamento/>}
        {tab==="projetos"     && <TabProjetos/>}
        {tab==="controle"     && (projetoSel ? <TabControle/> : <TabControleDash/>)}
        {tab==="fornecedores" && <TabFornecedores/>}
        {tab==="resumo"       && <TabResumo/>}
        {tab==="historico"    && <TabHistorico/>}
      </div>
      {showSiteModal && <SiteModal/>}
    </div>
  );
}
